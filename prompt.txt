    <script>
        // Timeline: Total 15s
        const TIMELINE = [
            { id: 'f1', duration: 3000 },  // Hook (3s)
            { id: 'f2', duration: 3000 },  // Buildup (3s)
            { id: 'f3', duration: 4000 },  // Reveal (4s)
            { id: 'f4', duration: 2500 },  // Message (2.5s)
            { id: 'f5', duration: 2500 }   // CTA (2.5s)
        ];

        function startTimeline() {
            let elapsed = 0;
            TIMELINE.forEach(step => {
                setTimeout(() => {
                    document.querySelectorAll('.frame').forEach(f => f.classList.remove('active'));
                    document.getElementById(step.id).classList.add('active');
                }, elapsed);
                elapsed += step.duration;
            });
        }

        // Particles
        function createParticle() {
            const p = document.createElement('div');
            p.className = 'particle';
            const size = Math.random() * 8 + 4;
            p.style.width = size + 'px';
            p.style.height = size + 'px';
            p.style.left = Math.random() * 100 + '%';
            p.style.bottom = '0';
            p.style.animation = `rise ${Math.random() * 3 + 3}s linear forwards`;
            
            const colors = ['var(--gold)', 'var(--cyan)', '#fff'];
            p.style.background = colors[Math.floor(Math.random() * colors.length)];
            
            document.getElementById('particles').appendChild(p);
            setTimeout(() => p.remove(), 6000);
        }

        setInterval(createParticle, 150);
